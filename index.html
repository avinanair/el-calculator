function calculateEL() { const from = new Date(document.getElementById('fromDate').value); const to = new Date(document.getElementById('toDate').value); const other = parseInt(document.getElementById('otherLeave').value) || 0; if (isNaN(from) || isNaN(to)) { alert("Please select both dates!"); return; } const diffTime = to - from; const totalDays = Math.ceil(diffTime / (1000 * 60 * 60 * 24)) + 1; const netWorkingDays = totalDays - other; if (netWorkingDays < 0) { alert("Leaves taken cannot exceed total days!"); return; } // Logic for "X - Y/11" format const fullDays = Math.floor(netWorkingDays / 11); const remainderDays = netWorkingDays % 11; let displayResult = ""; if (remainderDays === 0) { displayResult = fullDays; // Just show the number if it's exact } else { displayResult = `${fullDays} - ${remainderDays}/11`; } document.getElementById('result-area').style.display = 'block'; document.getElementById('elResult').innerText = displayResult; document.getElementById('breakdown').innerText = `Total Net Days: ${netWorkingDays}`; }
